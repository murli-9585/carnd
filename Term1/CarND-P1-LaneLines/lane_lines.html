Project: Finding Lane Lines on the Road¶

In this project, you will use the tools you learned about in the lesson to identify lane lines on the road. You can develop your pipeline on a series of individual images, and later apply the result to a video stream (really just a series of images). Check out the video clip "raw-lines-example.mp4" (also contained in this repository) to see what the output should look like after using the helper functions below.

Once you have a result that looks roughly like "raw-lines-example.mp4", you'll need to get creative and try to average and/or extrapolate the line segments you've detected to map out the full extent of the lane lines. You can see an example of the result you're going for in the video "P1_example.mp4". Ultimately, you would like to draw just one line for the left side of the lane, and one for the right.

import math

def grayscale(img):
    """Applies the Grayscale transform
    This will return an image with only one color channel
    but NOTE: to see the returned image as grayscale
    (assuming your grayscaled image is called 'gray')
    you should call plt.imshow(gray, cmap='gray')"""
    return cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
    # Or use BGR2GRAY if you read an image with cv2.imread()
    #return cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    
def canny(img, low_threshold, high_threshold):
    """Applies the Canny transform"""
    return cv2.Canny(img, low_threshold, high_threshold)

def gaussian_blur(img, kernel_size):
    """Applies a Gaussian Noise kernel"""
    return cv2.GaussianBlur(img, (kernel_size, kernel_size), 0)

def region_of_interest(img, vertices):
    """
    Applies an image mask.
    
    Only keeps the region of the image defined by the polygon
    formed from `vertices`. The rest of the image is set to black.
    """
    #defining a blank mask to start with
    mask = np.zeros_like(img)   
    
    #defining a 3 channel or 1 channel color to fill the mask with depending on the input image
    if len(img.shape) > 2:
        channel_count = img.shape[2]  # i.e. 3 or 4 depending on your image
        ignore_mask_color = (255,) * channel_count
    else:
        ignore_mask_color = 255
    
    #filling pixels inside the polygon defined by "vertices" with the fill color    
    cv2.fillPoly(mask, vertices, ignore_mask_color)
    
    #returning the image only where mask pixels are nonzero
    masked_edges = cv2.bitwise_and(img, mask)
    return masked_edges

def _update_lines(lines, delta):
    """
    Does cleanup and updates lines with slope and y-intercept.
    Return value format [[x1,y1,x2,y2,slope,y-intercept,line_len], [..]]
    """
    
    updated_lines = []
    for line in lines:
        for x1,y1,x2,y2 in line:
            
            # Avoid full vertical and horizantal lines.
            x_diff = x2 - x1
            y_diff = y2 - y1
            if not (x_diff & y_diff):
                continue
                
            # calculate slope.
            slope = y_diff/x_diff
            y_intercept = y1 - (slope*x1)
            line_len = math.sqrt((x2-x1)**2 + (y2-y1)**2)
            # Remove any flat slopes
            if -delta < slope < delta:
                continue
            updated_lines.append([x1, y1, x2, y2, slope, y_intercept, line_len])
        
    return updated_lines

def calculate_averages(lines):
    """
    Calculate avergage slope and average y-intercepts based on long lines.
    """
    
    big_lines = get_large_lines(lines)
    num_items = len(big_lines)
    result_lines = []
    if not num_items:
        return num_items
        
    avg_slope = sum([line[4] for line in big_lines])/num_items
    # Update y-intercept and avg_slope for all
    for line in big_lines:
        y_intercept = line[1] - (avg_slope*line[0])
        result_lines.append([line[0], line[1], line[2], line[3], avg_slope, y_intercept, line[6]])
    return result_lines
    
def split_lines(lines):
    """
    Splits the lines to pos and neg lines based on the slope.
    Takes updated line as input [[x1,y1,x2,y2,slope,y-intercept,line_len], [..]]
    """
    
    pos_lines = []
    neg_lines = []
    
    for line in lines:
        slope = line[4]
        if slope > 0:
            pos_lines.append(line)
        else:
            neg_lines.append(line)
    return pos_lines, neg_lines

def get_large_lines(lines):
    """
    Return large lines form the given set of lines.
    Expected modified lines.
    """
    
    # Sort on line length
    lines_sorted = sorted(lines, key=lambda x:x[6])
    # Half of lines.
    num_items = int(len(lines_sorted)/2)
    large_lines = lines_sorted[-num_items:]
    return large_lines
    
def extrapolate_lines(img, p_big_lines, n_big_lines):
    """
    Extrapolate based on average weights of large/big lines.
    """
    
    # Calculate average Y-Intercept
    p_count, n_count = len(p_big_lines), len(n_big_lines)
    p_avg_yintercept = sum([line[5] for line in p_big_lines])/p_count
    n_avg_yintercept = sum([line[5] for line in n_big_lines])/n_count
    # Y1 should be same for both which is the bottom of shape.
    y1 = img.shape[0]
    
    # Average slope is setup for every line.
    p_avg_slope = p_big_lines[0][4]
    n_avg_slope = n_big_lines[0][4]
    
    # X = y-b/m; 
    x1_p = (y1-p_avg_yintercept)/p_avg_slope
    x1_n = (y1-n_avg_yintercept)/n_avg_slope
    
    # Aim at ~60+% of image; Could be done better.
    y2 = img.shape[0]/1.57
    # Big extrapolation leads to line intersection on speed. 
    #y2_min_pos = min([line[3] for line in p_big_lines])
    #y2_min_neg = min([line[3] for line in n_big_lines])
    #y2 = min(y2_min_pos, y2_min_neg)
    x2_p = (y2-p_avg_yintercept)/p_avg_slope
    x2_n = (y2-n_avg_yintercept)/n_avg_slope
    
    # Create array lines and return.
    connected_lines = np.array([[[x1_p, y1, x2_p, y2], [x1_n, y1, x2_n, y2]]], dtype=np.int32)
    return connected_lines

def draw_lines(img, lines, color=[255, 0, 0], thickness=8):
    """
    NOTE: this is the function you might want to use as a starting point once you want to 
    average/extrapolate the line segments you detect to map out the full
    extent of the lane (going from the result shown in raw-lines-example.mp4
    to that shown in P1_example.mp4).  
    
    Think about things like separating line segments by their 
    slope ((y2-y1)/(x2-x1)) to decide which segments are part of the left
    line vs. the right line.  Then, you can average the position of each of 
    the lines and extrapolate to the top and bottom of the lane.
    
    This function draws `lines` with `color` and `thickness`.    
    Lines are drawn on the image inplace (mutates the image).
    If you want to make the lines semi-transparent, think about combining
    this function with the weighted_img() function below
    """
    # First iterate over lines and sort left/right.
    
    # Delta is used to remove noise. Assuming Average mean is slope +- delta.
    # Delta is how much a lane could turn from a average slope of a lane.
    # This can be further given a formula for slope. 
    # XXXMM: Can it be more than Sin(30)?
    delta = 0.4
    
    lines_addl_info = _update_lines(lines, delta)
    pos_lines, neg_lines = split_lines(lines_addl_info)
    pos_lines_avg = calculate_averages(pos_lines)
    neg_lines_avg = calculate_averages(neg_lines)
    draw = True
    if not (pos_lines_avg and neg_lines_avg):
        draw = False
    else:
        lines_exp = extrapolate_lines(img, pos_lines_avg, neg_lines_avg)

    # Draw Lines 
    if draw:
        for line in lines_exp:
            for x1,y1,x2,y2 in line:
                cv2.line(img, (x1, y1), (x2, y2), color, thickness)

def hough_lines(img, rho, theta, threshold, min_line_len, max_line_gap):
    """
    `img` should be the output of a Canny transform.
        
    Returns an image with hough lines drawn.
    """
    lines = cv2.HoughLinesP(img, rho, theta, threshold, np.array([]), minLineLength=min_line_len, maxLineGap=max_line_gap)
    line_img = np.zeros((img.shape[0], img.shape[1], 3), dtype=np.uint8)
    draw_lines(line_img, lines)
    return line_img

# Python 3 has support for cool math symbols.

def weighted_img(img, initial_img, =0.8, =1., =0.):
    """
    `img` is the output of the hough_lines(), An image with lines drawn on it.
    Should be a blank image (all black) with lines drawn on it.
    
    `initial_img` should be the image before any processing.
    
    The result image is computed as follows:
    
    initial_img *  + img *  + 
    NOTE: initial_img and img must be the same shape!
    """
    return cv2.addWeighted(initial_img, , img, , )

def draw_lanes(img): 
    """ 
    Given a input image process it to draw lanes by:
    a. Convert to grey scale; Reduce size 
    b. Apply Gaussian smoothing; Reduce noise 
    c. Create canny edges; Give structure 
    d. Mask the edges; Remove unwanted areas 
    e. Apply hough transform on masked polygon/edges 
    f. Combine line image on original img. and return line edges. 
    """
    # Convert the image to grey scale so it works for all color lines.
    gray_img = grayscale(img)

    # Remove noise before further processing; uses Gaussian smoothing
    kernel_size = 5
    gray_blur = gaussian_blur(gray_img, kernel_size)

    # Extract structural information using canny edge
    low_threshold = 60
    high_threshold = 180
    canny_edges = canny(gray_blur, low_threshold, high_threshold)

    # Mask everything else apart from the selected polygon - Quadrilateral.

    # The camera mount point should be the same irrespective of the lane
    # the picture was taken. Constants should be fine here.
    img_shape = img.shape
    vertices = np.array([[(110,img_shape[0]), (480, 320), (490, 320),
                (img_shape[1],img_shape[0])]], dtype=np.int32)
    
    mask_edges = region_of_interest(canny_edges, vertices)

    # Apply Hough transform on masked image.
    # Values straight from quiz.
    rho = 2 # distance resolution in pixels of the Hough grid
    theta = np.pi/180 # angular resolution in radians of the Hough grid
    threshold = 15   # minimum number of votes (intersections in Hough grid cell)
    min_line_length = 20 #minimum number of pixels making up a line
    max_line_gap = 40    # maximum gap in pixels between connectable line segments
    line_img = hough_lines(mask_edges, rho, theta, threshold, min_line_length, max_line_gap)

    # Overlay line image to original.
    overlay_img = weighted_img(line_img, img)
    return overlay_img
